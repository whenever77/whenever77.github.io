nums = [0,0,1]
numlen = len(nums)
i = 0
j = numlen - 1

while (i <= j):
while (nums[i] > 0):
i += 1
if i > numlen - 1:
break
while (nums[j] == 0):
j -= 1
if j < 0:
break

if i < j:
    for ii in range(i, j):
        nums[ii] = nums[ii + 1]
    nums[j] = 0
i += 1
j -= 1

print(nums)
